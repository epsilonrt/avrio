494 __STATIC_ALWAYS_INLINE (eTwiStatus eTwiSetSpeed (uint16_t usSpeed)) {
495  uint16_t usRatio;
496 
497 #ifdef TWPS0
498  TWSR &= ~(_BV (TWPS0) | _BV (TWPS1));
499 #endif
500 
501  usRatio = ((AVRIO_CPU_FREQ / 1000UL) / usSpeed);
502  if ((usRatio < 16) || (usRatio > (255 * 2 + 16))) {
503 
504  return TWI_ERROR_INVALID_SPEED;
505  }
506  TWBR = (uint8_t) (usRatio - 16) / 2;
507  return TWI_SUCCESS;
