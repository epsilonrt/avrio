444 vAfskInit (uint8_t ucMode) {
445 
446  memset (&af, 0, sizeof(af));
447  af.ucMode = ucMode;
448  vAfskDebugInit();
449  vAfskHwInit();
450 
451 #ifndef AFSK_TX_DISABLE
452  vQueueSetBuffer (&af.xTxFifo, af.ucTxBuffer, sizeof(af.ucTxBuffer));
453  af.usTxPhaseInc = AFSK_MARK_INC;
454  vAfskHwDacWrite (AFSK_DAC_VALUE(128)); // sin(0) -> VREF_DAC/2
455 #endif
456 
457 #ifndef AFSK_RX_DISABLE
458  vQueueSetBuffer (&af.xRxFifo, af.ucRxBuffer, sizeof(af.ucRxBuffer));
459  vQueueSetBuffer (&af.xSampleFifo, (uint8_t *)af.iSampleBuffer, sizeof(af.iSampleBuffer));
460 
461  /* Fill sample FIFO with 0 */
462  for (int i = 0; i < AFSK_SAMPLES_PER_BIT / 2; i++)
463  vQueuePush(&af.xSampleFifo, 0);
464 #endif
465 }
